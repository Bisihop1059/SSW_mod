abandon_colony_tag = {
	if = {	
		limit = {
			is_subject = yes
		}
		if = {
			limit = {
				original_tag = KOR
				is_puppet_of = JAP
			}
			set_cosmetic_tag = KOR_JAP
		}
		if = {
			limit = {
				original_tag = PAO
				is_puppet_of = JAP
			}
			set_cosmetic_tag = PAO_JAP
		}
		else = {	drop_cosmetic_tag = yes	}
	}
	else = {	drop_cosmetic_tag = yes	}
}

is_abandoned_colony_tag = {
	if = {
		limit = {
			tag = ROOT
		}
		ROOT = {
			abandon_colony_tag = yes
		}
	}
}
