on_actions = {
	on_annex = {	#国家を併合した時
		effect = {
			every_country = {
				limit = {
					original_colony_tag = yes
					is_puppet = no
				}
				abandon_colony_tag = yes
			}
		}
	}
	on_subject_free = {	#傀儡国が独立した時
		effect = {
			if = {
				limit = {
					original_colony_tag = yes
				}
				abandon_colony_tag = yes
			}
		}
	}
	on_release_as_free = {	#独立国として解放した時
		effect = {
			if = {
				limit = {	original_colony_tag = yes	}
				abandon_colony_tag = yes
			}
		}
	}

on_actions = {
	on_annex = {
		effect = {
			if = {
				limit = {
					AND = {
						tag = CZE
						CZE = {
							owns_state = 70
							owns_state = 71
						}
						SLO = {	exists = no	}
					}
				}
				set_cosmetic_tag = CZEandSLO
			}
			if = {
				limit = {
					AND = {
						tag = MEN
						MEN = {
							owns_state = 330
						}
						MON = {	exists = no	}
					}
				}
				set_cosmetic_tag = Great_Mongol
			}
		}
	}
	on_release_as_free = {
		effect = {
			if = {
				limit = {
					AND = {
						tag = MEN
						MON = {	exists = yes	}
					}
				}
				MEN = {	drop_cosmetic_tag = yes	}
			}
		}
	}
	on_release_as_puppet = {
		effect = {
			if = {
				limit = {
					AND = {
						tag = MEN
						MON = {	exists = yes	}
					}
				}
				MEN = {	drop_cosmetic_tag = yes	}
			}
			if = {
				limit = {
					AND = {
						tag = SPR
						is_subject = yes
						OVERLORD = {
							OR = {
								has_government = progressive_communism
								has_government = classical_communism
								has_government = social_democracy
								has_government = social_liberalism
							}
						}
					}
				}
				OVERLORD = { set_country_flag = puppet_spa }
				SPA = { annex_country = { target = SPR transfer_troops = yes } }
				every_state = {
					limit = {
						is_core_of = SPR
						NOT = { is_core_of = SPA }
					}
					add_core_of = SPA
				}
				every_state = {
					limit = { is_core_of = SPR }
					remove_core_of = SPR
				}
				random_country = {
					limit = { has_country_flag = puppet_spa }
					puppet = SPA
					clr_country_flag = puppet_spa
				}
			}
		}
	}
	on_puppet = {
		effect = {
			if = {
				limit = {
					AND = {
						tag = MEN
						MON = {	exists = yes	}
					}
				}
				MEN = {	drop_cosmetic_tag = yes	}
			}
			if = {
				limit = {
					AND = {
						tag = SPR
						is_subject = yes
						OVERLORD = {
							OR = {
								has_government = progressive_communism
								has_government = classical_communism
								has_government = social_democracy
								has_government = social_liberalism
							}
						}
					}
				}
				OVERLORD = { set_country_flag = puppet_spa }
				SPA = { annex_country = { target = SPR transfer_troops = yes } }
				every_state = {
					limit = {
						is_core_of = SPR
						NOT = { is_core_of = SPA }
					}
					add_core_of = SPA
				}
				every_state = {
					limit = { is_core_of = SPR }
					remove_core_of = SPR
				}
				random_country = {
					limit = { has_country_flag = puppet_spa }
					puppet = SPA
					clr_country_flag = puppet_spa
				}
			}
		}
	}
}
